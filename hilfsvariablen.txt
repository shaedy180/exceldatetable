-- Δ € anzeigen, wenn im Slicer "Δ Umsatz" gewählt ist.
-- Sonderfall: Wenn NUR "Keine" gewählt ist, bleibt es leer.
[Δ € Show] :=
VAR sel        = VALUES('DeltaToggle'[Auswahl])
VAR isKeineOnly= COUNTROWS(sel) = 1 && NOT ISEMPTY(FILTER(sel; 'DeltaToggle'[Auswahl] = "Keine"))
VAR hasDelta€  = NOT ISEMPTY(FILTER(sel; 'DeltaToggle'[Auswahl] = "Δ Umsatz"))
RETURN IF( NOT isKeineOnly && hasDelta€; [Δ €]; BLANK() )
